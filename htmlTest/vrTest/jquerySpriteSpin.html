  <script type="text/javascript">
  $(function(){
    
    // initialize the spritespin
    $(".spritespin").spritespin({
      width     : 640,
      height    : 437,
      frames    : 34,
      framesX   : 6,
      rsX       : 3840,
      resY      : 2622,
      sense     : -1,
      source    : "/images/spritespin/bike6x6_big.jpg",
      animate   : false
    }).bind("onLoad", function(){
      var $this = $(this), data = $this.data("spritespin");
      data.stage.find(".detail").detach();       // remove previous details
      data.stage.prepend($(".details .detail")); // add current details
      data.stage.find(".detail").hide();         // hide current details
    }).bind("onFrame", function(){
      var $this = $(this), data = $this.data("spritespin");
      if (data.frame >= 31 || data.frame <= 4){
        data.stage.find(".detail.first").fadeIn()
      } else {
        data.stage.find(".detail.first").fadeOut()
      }
  
      if (data.frame >= 5 && data.frame <= 12){
        data.stage.find(".detail.second").fadeIn()
      } else {
        data.stage.find(".detail.second").fadeOut()
      }
  
      if (data.frame >= 19 && data.frame <= 28){
        data.stage.find(".detail.last").fadeIn()
      } else {
        data.stage.find(".detail.last").fadeOut()
      }
    });
    
    var details = [0, 8, 20];
    var currentDetail = 0;
  
    // button jumping to previous frame
    // get current frame and update spritespin to the previous frame
    $( "#prev" ).click(function(){
      currentDetail -= 1;
      if (currentDetail < 0){
        currentDetail = details.length - 1;
      }
      if (currentDetail >= details.length){
        currentDetail = 0;
      }
      $(".spritespin").spritespin("animateTo", details[currentDetail]);
    });
    
    // button jumping to next frame
    $( "#next" ).click(function(){
      currentDetail += 1;
      if (currentDetail < 0){
        currentDetail = details.length - 1;
      }
      if (currentDetail >= details.length){
        currentDetail = 0;
      }
      $(".spritespin").spritespin("animateTo", details[currentDetail]);
    });
  });
  </script>
